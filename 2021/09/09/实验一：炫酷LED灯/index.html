//LED闪烁灯
int LEDPin = 3;
//延迟时间
#define dTime 1000
void setup() {
  //设置led pin为输出脚
  pinMode(LEDPin, OUTPUT);
}

void loop() {
  //高低位轮换， 间隔1000ms
  digitalWrite(LEDPin, HIGH);
  delay(dTime);
  digitalWrite(LEDPin, LOW);
  delay(dTime);
}
