int LEDPin = 3;

void setup() {
  // put your setup code here, to run once:
  pinMode(LEDPin, OUTPUT);
}

void loop() {
  // put your main code here, to run repeatedly:
  int val;
  while (true) {
    //获取电压值， 0~5V : 0 ~ 1023
    val = analogRead(A0);
    if (val >= 1000) {
      digitalWrite(LEDPin, HIGH);
    } else {
      digitalWrite(LEDPin, LOW);
    }
  }
}

7. 扩展实验

上述实验中， 按键按下LED亮，要想LED一直亮着， 按键需一直按下。修改源码，实现当按键按下并放开后，LED任然会一直亮着

int LEDPin = 3;
bool w = false; //开关按下， w取反

void setup()
{
    pinMode(LEDPin, OUTPUT);
}

void loop()
{
    while (true)
    {
        //获取电压值， 0~5V : 0 ~ 1023
        if (analogRead(A0) >= 1000)
        {
            w = !w;
        }
        if (w)
        {
            digitalWrite(LEDPin, HIGH);
        }
        else
        {
            digitalWrite(LEDPin, LOW);
        }
        delay(10);
    }
}
