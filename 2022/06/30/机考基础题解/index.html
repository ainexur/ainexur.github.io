（1）. 输入处理（）：HJ5.进制转换

牛客链接

题意：

接受一个十六进制的数，输出该数值的十进制表示（int范围内的正整数）。

#include <iostream>
using namespace std;

int main () {
    int n;
    scanf("%x",&n);
    printf("%d", n);
    return 0;
}

解析：

利用c语言的输入格式%x可以直接接收十六进制的输入，自动转成int类型的十进制数，再将其值输出即可。

难度1颗星，主要考察基本的输入格式。

（3）快速排序：HJ3.明明的随机数

牛客链接

题意：

输入n个带重复的整数，按顺序输出去重后的数字

#include <iostream>
#include <algorithm>
using namespace std;

const int N = 1010;
int a[N];
int n;

void quick_sort(int l, int r) {
    if (l >= r) return;
    int x = a[(l + r) / 2];
    int i = l - 1, j = r + 1;
    while (i < j) {
        while (a[++i] < x);
        while (a[--j] > x);
        if (i < j) swap(a[i], a[j]);
    }
    quick_sort(l, j);
    quick_sort(j + 1, r);
}
int main () {
    scanf("%d", &n);
    for (int i = 0; i < n; i ++) {
        scanf("%d", &a[i]);
    }
    
    quick_sort(0, n - 1);
    
    for (int i = 0; i < n; i ++) {
        if (a[i] != a[i + 1])
            printf("%d\n", a[i]);
    }
    
    return 0;
}

解析：

考察手写快排等排序。难度两颗星，若平时没注意练习快排，刚刚就凉了一次。

(5) 递归：NC68.跳台阶

class Solution {
public:
    int f[50];
    int jumpFloor(int number) {
        f[0] = f[1] = 1;
        for (int i = 2; i <= number; i ++)
            f[i] = f[i - 1] + f[i - 2];
        return f[number];
    }
};

class Solution {
public:
    int f[50];
    int jumpFloor(int number) {
        int a = 0, b = 1, c = 0;
        while (number --) {
            c = a + b;
            a = b;
            b = c;
        }
        return b;
    }
};

(1) HJ17.坐标移动

模拟题，细心即可

#include <iostream>
#include <cstring>
using namespace std;

char s[10010];
int x = 0, y = 0;
char dir[] = "ADWS";

void check(int l, int r) {
    bool isalpha = false;
    for (int i = 0; i < 4; i ++)
        if (dir[i] == s[l])
            isalpha = true;
    if (!isalpha) return;
    int num = 0;
    for (int i = l + 1; i <= r; i ++) {
        if (!isdigit(s[i])) return;
        num = num * 10 + s[i] - '0';
    }
    switch(s[l]) {
        case 'A': x -= num;break;
        case 'D': x += num;break;
        case 'W': y += num;break;
        case 'S': y -= num;break;
        default:break;
    }
}
int main() {
    scanf("%s", s);
    int n = strlen(s);
    int i = 0;
    while (i < n) {
        if (s[i] == ';')  {
            ++ i;
            continue;
        }
        int l = i, r = i;
        while (s[r] != ';') ++ r;
        check(l, r - 1);
        i = r;
    }
    cout << x << ',' << y;
    return 0;
}

(2) HJ20.密码验证合格程序

模拟题，细心即可

#include <iostream>

using namespace std;

bool check(string &s) {
    if (s.size() < 8) {
        return false;
    }
    int a[4] = {0};
    for (char c : s) {
        if (isupper(c)) a[0] ++;
        else if (islower(c)) a[1] ++;
        else if (isdigit(c)) a[2] ++;
        else a[3] ++;
    }
    int cnt = 0;
    for (int i = 0; i < 4; i ++) cnt += (a[i] > 0);
    if (cnt < 3) return false;
    for (int i = 0, n = s.size(); i < n; i ++) {
        for (int j = i + 1; j < n; j ++) {
            int l = i, r = j;
            while (s[l] == s[r]) ++l, ++r;
            if (l - i > 2) return false;
        }
    }
    return true;
}

int main(){
    string s;
    while (cin >> s) {
        cout << (check(s) ? "OK" : "NG") << endl;
    }
    return 0;
}

(4) *HJ33.整数与IP地址间的转换

#include <iostream>

using namespace std;

void cal1(const string& str) {
    long long ans = 0;
    long long cur = 0;
    for (char c : str) {
        if (c == '.') {
            ans <<= 8;
            ans += cur;
            cur = 0;
        } else {
            cur = cur * 10 + c - '0';
        }
    }
    ans <<= 8;
    ans += cur;
    cout << ans << endl;
}

void cal2(const string& str) {
    long long num = 0;
    for (char c : str) num = num * 10 + c - '0';
    string ans;
    while (num) {
        int t = num & 0xFF;
        ans = '.' + to_string(t) + ans;
        num >>= 8;
    }
    ans = ans.substr(1, ans.size() - 1);
    cout << ans << endl;
}
int main() {
    string str;
    cin >> str;
    cal1(str);
    cin >> str;
    cal2(str);
    return 0;
}

(5) HJ101.输入整型数组和排序标识

手撸快排

#include <iostream>

using namespace std;
const int N = 1010;
int n;
int a[N];
int flag;

bool check (int x, int y) {
    return flag == 0 ? x < y : x > y;
}
void quick_sort(int l, int r) {
    if (l >= r) return;
    int mid = a[(l + r) >> 1], i = l - 1, j = r + 1;
    while (i < j) {
        while (check(a[++i],mid));
        while (check(mid, a[--j]));
        if (i < j) swap(a[i], a[j]);
    }
    
    quick_sort(l, j);
    quick_sort(j + 1, r);
}
int main() {
    cin >> n;
    for (int i = 0; i < n; i ++) cin >> a[i];
    cin >> flag;
    
    quick_sort(0, n - 1);
    for (int i = 0; i < n; i ++)
        cout << a[i] << " ";
    return 0;
}

(3) HJ27.查找兄弟单词

字符串比较，以及排序基本运用

#include <iostream>
#include <vector>
#include <algorithm>
#include <unordered_map>
using namespace std;

bool check(string a, string b) {
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());
    return a == b;
}
int main() {
    int n; cin >> n;
    vector<string> arr;
    while (n --) {
        string s; cin >> s;
        arr.push_back(s);
    }
    
    string str; 
    int k;
    cin >> str >> k;
    vector<string> ans;
    for (string s : arr) {
        if ( s != str && check(s, str))
            ans.push_back(s);
    }
    
    cout << ans.size() << endl;
    sort(ans.begin(), ans.end());
    if (ans.size() >= k) cout << ans[k - 1];
    return 0;
}
